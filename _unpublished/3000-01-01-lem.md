interface Monad m => MonadLEM (m : Type -> Type) where

interface Monad m => MonadImpliesToOr (m : Type -> Type) where

MonadLEM m => MonadCont m where
    callCC cont = mlem >>= either pure go
        where
        go not_p = cont (\x => map absurd (not_p x))
